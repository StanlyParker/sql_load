 SELECT
        skills_demand.skill_id,
        skills_demand.skills,
        demand_count,
        average_salary.average_salary
FROM skills_demand
WHERE demand_count > 10
INNER JOIN average_salary 
ON average_salary.skill_id = skills_demand.skill_id
ORDER BY demand_count DESC,
         average_salary DESC
LIMIT 25;
